import React, { useState } from 'react'

function SearchSection() {
  const [hallCode, setHallCode] = useState("")
  const [result, setResult] = useState(null)

  const handleSearch = () => {
    // This will be replaced with real data later
    if (hallCode.toLowerCase() === "csc101") {
      setResult({
        name: "Computer Science Hall 101",
        image: "/images/csc101.jpg",
        direction: "Walk straight from main gate, turn right after the library, itâ€™s 2nd building"
      })
    } else {
      setResult(null)
    }
  }

  return (
    <section className="search-section">
      <h1>Search for Hall</h1>
      <input
        type="text"
        placeholder="Enter hall code or name..."
        value={hallCode}
        onChange={(e) => setHallCode(e.target.value)}

      {result && (
